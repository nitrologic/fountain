	const raw = new Uint8Array(1024);
	let buffer="";
	while (true) {
	const n = await stream.read(raw);
	if (n === null) break;
	buffer+=decoder.decode(raw.subarray(0,n));
	echo("[" + tag + "]",buffer.trimEnd());
	buffer="";
	await flush();
	}
	}
